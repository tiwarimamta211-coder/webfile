const fmt = n => '₹' + (parseFloat(n)||0).toLocaleString('en-IN',{minimumFractionDigits:2,maximumFractionDigits:2});
const pct  = n => (parseFloat(n)||0).toFixed(2) + '%';

function getFY(){
  const d = new Date(); const m = d.getMonth()+1;
  const y = d.getFullYear();
  if(m>=4) return `${y}-${String(y+1).slice(2)}`;
  return `${y-1}-${String(y).slice(2)}`;
}

function getNextInvNo(){
  const fy = getFY();
  invCounter[fy] = (invCounter[fy]||0)+1;
  localStorage.setItem('acc_inv_counter',JSON.stringify(invCounter));
  return `INV/${fy}/${String(invCounter[fy]).padStart(4,'0')}`;
}

function isInter(state){ return (state||'').trim().toLowerCase() !== CONFIG.state.toLowerCase(); }

function calcGST(amount, rate, state){
  const base = parseFloat(amount)||0;
  const inter = isInter(state);
  const h = rate/2;
  if(inter) return {cgst:0,sgst:0,igst:+(base*rate/100).toFixed(2),rc:'Yes'};
  return {cgst:+(base*h/100).toFixed(2),sgst:+(base*h/100).toFixed(2),igst:0,rc:'No'};
}

function calcTDS(amount, section){
  const base = parseFloat(amount)||0;
  const rate = TDS_RATES[section]||0;
  return {rate, tds:+(base*rate/100).toFixed(2), net:+(base - base*rate/100).toFixed(2)};
}

function addTransaction(){
  const date   = document.getElementById('f_date').value;
  const type   = document.getElementById('f_type').value;
  const desc   = document.getElementById('f_desc').value.trim();
  const party  = document.getElementById('f_party').value.trim();

  const state  = document.getElementById('f_state').value;
  const hsn    = document.getElementById('f_hsn').value.trim();
  const tdsPart= document.getElementById('f_tds').value;
  const tds    = tdsPart.split('|')[0];

  if(!date||!desc||!party||isNaN(amount)||amount<=0){
    alertEl.innerHTML=`<div class="alert alert-err">⚠️ Sab required fields bharen aur amount positive hona chahiye.</div>`;
    return;
  }
  // Duplicate check
  const dup = TRANSACTIONS.find(t=>t.date===date&&t.desc===desc&&t.party===party&&t.amount===amount);

    alertEl.innerHTML=`<div class="alert alert-warn">⚠️ Yahi transaction already exist karti hai (duplicate).</div>`;

function renderMonthly(){

function generateInvoice(){
  const party  = document.getElementById('inv_party').value.trim();
  const gstin  = document.getElementById('inv_gstin').value.trim();
  const state  = document.getElementById('inv_state').value;
  if(!party){ alert('Client name bharen'); return; }

  const itemDivs = document.getElementById('invItems').children;
  const items = [];
  let subtotal = 0;
  for(const div of itemDivs){
    const inputs = div.querySelectorAll('input');
    const desc = inputs[0].value.trim();
    const qty  = parseFloat(inputs[1].value)||1;
    const rate = parseFloat(inputs[2].value)||0;
    if(desc) { items.push({desc,qty,rate,amt:qty*rate}); subtotal+=qty*rate; }
  }
  if(!items.length){ alert('Kam se kam ek item bharen'); return; }

  const inter = isInter(state);
  const gstRate = 18;
  let cgst=0,sgst=0,igst=0;
  if(inter){ igst=+(subtotal*gstRate/100).toFixed(2); }
  else { cgst=+(subtotal*gstRate/2/100).toFixed(2); sgst=cgst; }
  const grand = subtotal+cgst+sgst+igst;
  const invNo  = getNextInvNo();
  const invDate= new Date().toLocaleDateString('en-IN');

  const invData = {invNo,party,gstin,state,items,subtotal,cgst,sgst,igst,grand,date:invDate,inter};
  INVOICES.push(invData); saveInv();

  document.getElementById('invoicePreview').innerHTML = buildInvoiceHTML(invData);
}

function importFile(input){
  const file = input.files[0];
  if(!file) return;
  const res = document.getElementById('importResult');
  res.innerHTML=`<div class="alert alert-warn">⏳ File process ho rahi hai...</div>`;

  const reader = new FileReader();
  reader.onload = e => {
    try {
      const wb = XLSX.read(e.target.result, {type:'binary'});
      const ws = wb.Sheets[wb.SheetNames[0]];
      const data = XLSX.utils.sheet_to_json(ws);
      if(!data.length){ res.innerHTML=`<div class="alert alert-err">❌ File mein koi data nahi mila.</div>`; return; }

      const rename = col => {
        const l = col.trim().toLowerCase();
        if(['amount','amt','value','total'].includes(l)) return 'amount';
        if(['date','invoice date','bill date'].includes(l)) return 'date';
        if(['description','desc','item','particulars'].includes(l)) return 'desc';
        if(['type','category','head'].includes(l)) return 'type';
        if(['party','customer','vendor','client'].includes(l)) return 'party';
        if(['hsn','sac','hsn_sac','hsn/sac'].includes(l)) return 'hsn';
        if(['gst_rate','gst rate','gst%'].includes(l)) return 'gstRate';
        if(['state','party state'].includes(l)) return 'state';
        if(['tds','tds_section','tds section'].includes(l)) return 'tds';
        return col.toLowerCase();
      };

      let added=0, skipped=0, errors=[];
      data.forEach((row,idx)=>{
        const mapped={};
        Object.entries(row).forEach(([k,v])=>{ mapped[rename(k)]=v; });
        // Validation
        if(!mapped.amount || isNaN(parseFloat(mapped.amount))){ errors.push(`Row ${idx+2}: Invalid amount`); skipped++; return; }
        if(parseFloat(mapped.amount)<0){ errors.push(`Row ${idx+2}: Negative amount`); skipped++; return; }
        const type = mapped.type||'Income';
        if(!['Income','Expense','income','expense'].includes(type)){ errors.push(`Row ${idx+2}: Invalid type '${type}'`); skipped++; return; }

        const txn = {
          id: Date.now()+added,
          date: mapped.date ? String(mapped.date).slice(0,10) : today(),
          desc: mapped.desc||mapped.description||'Imported',
          type: type.charAt(0).toUpperCase()+type.slice(1).toLowerCase(),
          party: mapped.party||'Unknown',
          amount: parseFloat(mapped.amount),
          gstRate: parseInt(mapped.gstRate||18),
          state: mapped.state||CONFIG.state,
          hsn: mapped.hsn||'',
          tds: mapped.tds||'No TDS',
        };
        // Duplicate check
        const dup=TRANSACTIONS.find(t=>t.date===txn.date&&t.desc===txn.desc&&t.amount===txn.amount&&t.party===txn.party);
        if(dup){ errors.push(`Row ${idx+2}: Duplicate skipped`); skipped++; return; }
        TRANSACTIONS.push(txn); added++;
      });
      save();
      let msg=`<div class="alert alert-ok">✅ ${added} transactions import hue!</div>`;
      if(skipped) msg+=`<div class="alert alert-warn">⚠️ ${skipped} rows skip hue:</div>`;
      if(errors.length) msg+=`<div style="font-size:.83rem;color:var(--muted);margin-top:8px;">${errors.slice(0,5).join('<br/>')}</div>`;
      res.innerHTML=msg;
    } catch(err){
      res.innerHTML=`<div class="alert alert-err">❌ File read karne mein error: ${err.message}</div>`;
    }
  };
  reader.readAsBinaryString(file);
}

function exportSheet(type){
  let data=[]; let filename='';
  if(type==='all'){
    data=TRANSACTIONS.map(t=>({Date:t.date,Description:t.desc,Type:t.type,Party:t.party,Amount:t.amount,GST_Rate:t.gstRate,State:t.state,HSN_SAC:t.hsn,TDS_Section:t.tds}));
    filename='All_Transactions.xlsx';
  } else if(type==='gst'){
    data=TRANSACTIONS.map(t=>{
      const g=calcGST(t.amount,t.gstRate||18,t.state);
      return {Date:t.date,Description:t.desc,Party:t.party,State:t.state,
        Is_Inter_State:isInter(t.state)?'Yes':'No',Base_Amount:t.amount,
        CGST_Amt:g.cgst,SGST_Amt:g.sgst,IGST_Amt:g.igst,Total_GST:g.cgst+g.sgst+g.igst,Grand_Total:t.amount+g.cgst+g.sgst+g.igst,Reverse_Charge:g.rc};
    });
    filename='GST_Report.xlsx';
  } else if(type==='tds'){
    data=TRANSACTIONS.map(t=>{
      const td=calcTDS(t.amount,t.tds||'No TDS');
      return {Date:t.date,Description:t.desc,Party:t.party,Amount:t.amount,TDS_Section:t.tds,TDS_Rate:td.rate+'%',TDS_Amount:td.tds,Net_Payable:td.net};
    });
    filename='TDS_Register.xlsx';
  } else if(type==='monthly'){
    const months={};
    TRANSACTIONS.forEach(t=>{
      const m=t.date?t.date.slice(0,7):'Unknown';
      if(!months[m]) months[m]={inc:0,exp:0};
      if(t.type==='Income') months[m].inc+=t.amount;
      else months[m].exp+=t.amount;
    });
    data=Object.entries(months).sort().map(([m,v])=>{
      const profit=v.inc-v.exp;
      return {Month:m,Income:v.inc,Expense:v.exp,Net_Profit:profit,Profit_Margin_pct:(v.inc?(profit/v.inc*100).toFixed(2):0)+'%'};
    });
    filename='Monthly_Summary.xlsx';
  }
  const ws = XLSX.utils.json_to_sheet(data);
  const wb = XLSX.utils.book_new();
  XLSX.utils.book_append_sheet(wb,ws,type);
  XLSX.writeFile(wb,filename);
}
